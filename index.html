Getting Started with Canvas

Setting up canvas

							const canvas = document.getElementById('myCanvas');
							const ctx = canvas.getContext('2d');

Drawing a Rectangle

							
	// Draw a filled rectangle
	// fillRect(x, y, width, height)
	ctx.fillStyle = 'green';
	ctx.fillRect(50, 50, 100, 75);

	// Draw a rectangle with border
	ctx.strokeRect(150, 150, 50, 50);

	// Sets the pixels in an area to transparent black rgba(0,0,0,0)
	ctx.clearRect(50, 50, 30, 30);

Drawing a Line

							
				// Draw a line
				// moveTo(x, y), lineTo(x, y)
				ctx.lineWidth = 10;
				ctx.beginPath();
				ctx.moveTo(50, 140);
				ctx.lineTo(150, 60);
				ctx.lineTo(250, 140);
				ctx.closePath();
				ctx.stroke();

				//Draw a curve line
				ctx.bezierCurveTo(50, 50, 90, 10, 100, 50);

Drawing Arcs & Circles

							//Draw a circle
//ctx.arc(x, y, radius, startAngle, endAngle, anticlockwise);
ctx.beginPath();
ctx.arc(75, 75, 50, 0, 2 * Math.PI);
ctx.stroke();

//Draw a circular arc
//arcTo(x1, y1, x2, y2, radius)
ctx.arcTo(100, 100, 300, 150, 50);

//Draw an ellipse
//ellipse(x, y, radiusX, radiusY, rotation, startAng, endAng, counterclockwise)
ctx.ellipse(100, 100, 50, 75, Math.PI / 4, 0, 2 * Math.PI);

Making clips

							
					// Draw clip
					ctx.beginPath();
					ctx.arc(100, 75, 50, 0, Math.PI * 2);
					ctx.clip();

					// Draw stuff that gets clipped
					ctx.fillStyle = "blue";
					ctx.fillRect(0, 0, canvas.width, canvas.height);

Canvas Animation

Creating an Animation

							
let x = 0;

function draw() {
	// Clear the canvas
	ctx.clearRect(0, 0, canvas.width, canvas.height);

	// Draw a moving rectangle
	ctx.fillStyle = 'red';
	ctx.fillRect(x, 50, 50, 50);

	// Update the x position for the next frame
	x += 1;

	// Request the next animation frame
	requestAnimationFrame(draw);
}

// Start the animation
requestAnimationFrame(draw);
